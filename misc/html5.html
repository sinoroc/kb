<html>
 <head>
  <title>Title</title>
 </head>
 <body>
  <main>
   <h1>Title</h1>
   <article>
    <h2>Section</h2>
    <section>
     <h3>Subsection</h3>
     <p>Content</p>
    </section>
   </article>
  </main>
 </body>
</html>
